\newcommand\drawWheel[3]{
  \begin{tikzpicture}
    %\fill [white] (0, 0) circle (6);
    \draw [#1, line width=1.5pt] (0, 0) circle (1);
    \foreach \s in {1, ..., #2} {
      \draw [#1, line width=1.5pt] (90+360/#2*\s:1) -- (90+360/#2*\s:1.15);
    }
    \node [#1] at (0, 0) {\huge{#3}};
  \end{tikzpicture}
}
